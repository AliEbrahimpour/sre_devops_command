pre-increment

awk 'BEGIN { a = 10; b = ++a; printf "a = %d, b = %d\n", a, b }'

Pre-Decrement
awk 'BEGIN { a = 10; b = --a; printf "a = %d, b = %d\n", a, b }'

Post-Increment

awk 'BEGIN { a = 10; b = a++; printf "a = %d, b = %d\n", a, b }'

Post-Decrement

awk 'BEGIN { a = 10; b = a--; printf "a = %d, b = %d\n", a, b }'
